2.4 Boxplot summary

Code 
# pivot sw to have all metrics in a metrics col and values in a values col
sw_long <- sw %>% pivot_longer(cols = c(2:7), names_to = "metric", values_to = "value")
sw_long %>% ggplot(aes(x=metric, y=value))+geom_boxplot()
